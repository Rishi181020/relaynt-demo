def test_device_calibration():
    # Simulates a minor environment configuration drift (SEV 3)
    calibration_drift = 12.4
    max_tolerance = 5.0
    assert calibration_drift <= max_tolerance, (
        f"ConfigurationError: Calibration offset drift ({calibration_drift}%) "
        f"exceeded maximum tolerance ({max_tolerance}%). Run env recalibration."
    )

if __name__ == "__main__":
    test_device_calibration()

import json

def test_broken_auth_flow():
    # Intentionally invalid JSON string using single quotes instead of double quotes.
    bad_json_string = "{'status': 'authenticated', 'token_valid': true}"
    config = json.loads(bad_json_string)
    assert config["status"] == "authenticated"

if __name__ == "__main__":
    test_broken_auth_flow()
